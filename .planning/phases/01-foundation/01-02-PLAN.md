    3. **frontend/lib/api.ts** — API client module:
       ```typescript
       // Base URL for the FastAPI backend server
       const API_BASE_URL = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000";

       // Generic fetch wrapper that handles JSON responses and errors
       export async function apiFetch<T>(endpoint: string, options?: RequestInit): Promise<T> {
         const response = await fetch(`${API_BASE_URL}${endpoint}`, {
           headers: { "Content-Type": "application/json", ...options?.headers },
           ...options,
         });
         if (!response.ok) {
           throw new Error(`API error: ${response.status} ${response.statusText}`);
         }
         return response.json();
       }

       // Check if the backend API is healthy and responding
       export async function checkHealth(): Promise<{ status: string; app: string }> {
         return apiFetch("/");
       }
       ```

    4. **Update frontend/app/globals.css** — Ensure the color palette is neutral/clean:
